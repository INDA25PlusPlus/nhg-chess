24/// Checks whether a pawn move reaches its promotion rank.
25///
26/// # Arguments
27///
28/// * `to` — The destination square index (0–63).
29/// * `piece` — The pawn being moved.
30///
31/// # Returns
32///
33/// * `true` if the pawn lands on its promotion rank (rank 8 for White, rank 1 for Black).
34/// * `false` otherwise.
35pub fn is_pawn_promotion(to: u8, piece: Piece) -> bool {
36    match piece {
37        Piece::Pawn(Color::White) => to >= 56, // rank 8
38        Piece::Pawn(Color::Black) => to < 8,   // rank 1
39        _ => false,
40    }
41}

43/// Returns the possible promotion pieces for a pawn that reaches the last rank.
44///
45/// By chess rules, promotion is mandatory, and the pawn can be promoted to
46/// a queen, rook, bishop, or knight of the same color.
47///
48/// # Arguments
49///
50/// * `piece` — The pawn being promoted.
51///
52/// # Returns
53///
54/// A `Vec<Piece>` containing the possible promotion outcomes.
55pub fn valid_pawn_promotions(piece: Piece) -> Vec<Piece> {
56    match piece {
57        Piece::Pawn(Color::White) => vec![
58            Piece::Queen(Color::White),
59            Piece::Rook(Color::White),
60            Piece::Bishop(Color::White),
61            Piece::Knight(Color::White),
62        ],
63        Piece::Pawn(Color::Black) => vec![
64            Piece::Queen(Color::Black),
65            Piece::Rook(Color::Black),
66            Piece::Bishop(Color::Black),
67            Piece::Knight(Color::Black),
68        ],
69        _ => vec![],
70    }
71}

73/// Generates *pseudo-legal* castling moves for a king, if available, based on castling rights
74/// and whether the squares between the king and rook are empty.
75///
76/// Castling is only possible if:
77/// - The king and the involved rook have not previously moved.
78/// - The path between the king and rook is empty.
79///
80/// # Arguments
81///
82/// * `from` — The square index of the king (should be `4` for White, `60` for Black).
83/// * `piece` — The king piece (must be `Piece::King(Color::White|Black)`).
84/// * `position` — The current position, including castling rights and occupancy.
85///
86/// # Returns
87///
88/// A vector of possible castling moves. May be empty if no castling is available.
89pub fn castling_moves(from: u8, piece: Piece, position: &Position) -> Vec<Move> {
90    let mut moves = Vec::new();
91    let cr = &position.castling_rights;
92    let (own_index, enemy_index) = piece_indexes(piece);
93
94    match piece.color() {
95        Color::White => {
96            if from == 4 { 
97                // kingside
98                if !cr.white_king_moved && !cr.white_kingside_rook_moved {
99                    let empty = !(position.bb_sides[own_index].0 | position.bb_sides[enemy_index].0);
100                    if (empty & (1<<5 | 1<<6)) == (1<<5 | 1<<6) {
101                        moves.push(Move { from, to: 6, piece, promoted_from_pawn: false });
102                    }
103                }
104                // queenside
105                if !cr.white_king_moved && !cr.white_queenside_rook_moved {
106                    let empty = !(position.bb_sides[own_index].0 | position.bb_sides[enemy_index].0);
107                    if (empty & (1<<1 | 1<<2 | 1<<3)) == (1<<1 | 1<<2 | 1<<3) {
108                        moves.push(Move { from, to: 2, piece, promoted_from_pawn: false });
109                    }
110                }
111            }
112        }
113        Color::Black => {
114            if from == 60 { 
115                // kingside
116                if !cr.black_king_moved && !cr.black_kingside_rook_moved {
117                    let empty = !(position.bb_sides[own_index].0 | position.bb_sides[enemy_index].0);
118                    if (empty & (1<<61 | 1<<62)) == (1<<61 | 1<<62) {
119                        moves.push(Move { from, to: 62, piece, promoted_from_pawn: false,});
120                    }
121                }
122                // queenside
123                if !cr.black_king_moved && !cr.black_queenside_rook_moved {
124                    let empty = !(position.bb_sides[own_index].0 | position.bb_sides[enemy_index].0);
125                    if (empty & (1<<57 | 1<<58 | 1<<59)) == (1<<57 | 1<<58 | 1<<59) {
126                        moves.push(Move { from, to: 58, piece, promoted_from_pawn: false, });
127                    }
128                }
129            }
130        }
131    }
132
133    moves
134}
