5/// Represents a chess move on the board.
6#[derive(Debug, Clone, Copy, Eq)]
7pub struct Move {
8    /// The starting square index (0–63).
9    pub from: u8,
10    /// The target square index (0–63).
11    pub to: u8,
12    /// The piece being moved (including color).
13    pub piece: Piece,
14    /// Whether this move resulted from a pawn promotion.
15    pub promoted_from_pawn: bool,
16}
17
18impl PartialEq for Move {
19    /// Equality ignores the piece type and promotion flag.
20    /// Only `from`, `to`, and `piece.color()` must match.
21    fn eq(&self, other: &Self) -> bool {
22        self.from == other.from && self.to == other.to && self.piece.color() == other.piece.color()
23    }
24}

26/// Returns all pseudo-legal moves for the given piece on the given square.
27///
28/// Delegates to the appropriate move generator based on the piece type.
29/// Does not check for king safety.
30///
31/// # Arguments
32/// * `from` - The square index (0–63) of the piece.
33/// * `piece` - The piece being moved.
34/// * `position` - Current board state.
35///
36/// # Returns
37/// A vector of pseudo-legal moves.
38pub fn valid_moves(from: u8, piece: Piece, position: &Position) -> Vec<Move> {
39    match piece {
40        Piece::Knight(_) => valid_knight_moves(from, piece, position),
41        Piece::Pawn(_) => valid_pawn_moves(from, piece, position),
42        Piece::Bishop(_) => valid_bishop_moves(from, piece, position),
43        Piece::Rook(_) => valid_rook_moves(from, piece, position),
44        Piece::Queen(_) => valid_queen_moves(from, piece, position),
45        Piece::King(_) => valid_king_moves(from, piece, position),
46    }
47}

50/// Returns side indices for a piece.
51///
52/// The board maintains bitboards for each side.  
53/// This helper returns `(friendly_index, enemy_index)` as `(0,1)` if the piece is white,
54/// and `(1,0)` if the piece is black.
55///
56/// # Arguments
57/// * `piece` - The piece being checked.
58///
59/// # Returns
60/// Tuple `(friendly_index, enemy_index)`.
61pub fn piece_indexes(piece: Piece) -> (usize, usize) {
62    match piece {
63        Piece::Bishop(Color::White)
64        | Piece::Knight(Color::White)
65        | Piece::Pawn(Color::White)
66        | Piece::Rook(Color::White)
67        | Piece::Queen(Color::White)
68        | Piece::King(Color::White) => (0, 1),
69
70        Piece::Bishop(Color::Black)
71        | Piece::Knight(Color::Black)
72        | Piece::Pawn(Color::Black)
73        | Piece::Rook(Color::Black)
74        | Piece::Queen(Color::Black)
75        | Piece::King(Color::Black) => (1, 0),
76    }
77}

79/// Generates pseudo-legal knight moves from a square.
80///
81/// Knights can jump over pieces.  
82/// This function checks board boundaries and excludes landing on friendly pieces.
83pub fn valid_knight_moves(from: u8, piece: Piece, position: &Position) -> Vec<Move> {
84    // see knight-offset.jpg
85    let knight_offsets: [i8; 8] = [17, 15, 10, 6, -17, -15, -10, -6];
86    let mut moves = Vec::new();
87
88    // Gives column as a value of an integer, i.e. 0 = A, 1 = B, .... 7 = H
89    let from_column = (from % 8) as i8;
90    // Gives row as value of an integer, i.e. 0 = row 1, 1 = row 2 ... 7 = row 8
91    let from_row = (from / 8) as i8;
92
93    for &offset in &knight_offsets {
94        let target = from as i8 + offset;
95
96        // stay inside board
97        if target < 0 || target > 63 {
98            continue;
99        }
100
101        let target_column = (target % 8) as i8;
102        let target_row = (target / 8) as i8;
103
104        // horse cannot move greater than 2 squares in one direction
105        if (from_column - target_column).abs() > 2 || (from_row - target_row).abs() > 2 {
106            continue;
107        }
108
109        let spotlight = 1u64 << target;
110
111        // cannot land on friendly piece
112        // move this function elsewhere - MERGE WITH CHECK_COLOR
113        let side_index = match piece {
114            Piece::Pawn(c)
115            | Piece::Rook(c)
116            | Piece::Knight(c)
117            | Piece::Bishop(c)
118            | Piece::Queen(c)
119            | Piece::King(c) => match c {
120                Color::White => 0,
121                Color::Black => 1,
122            },
123        };
124
125        if (position.bb_sides[side_index].0 & spotlight) != 0 {
126            continue;
127        }
128
129        moves.push(Move {
130            from,
131            to: target as u8,
132            piece,
133            promoted_from_pawn: false,
134        });
135    }
136    moves
137}

139/// Generates pseudo-legal bishop moves from a square.
140///
141/// Bishops move diagonally until blocked.  
142/// Captures are included, but the ray stops at the first blocking piece.
143pub fn valid_bishop_moves(from: u8, piece: Piece, position: &Position) -> Vec<Move> {
144    let mut moves = Vec::new();
145
146    let (friendly_index, enemy_index) = piece_indexes(piece);
147
148    let directions: [i8; 4] = [7, 9, -7, -9];
149    for &dir in &directions {
150        let mut target = from as i8;
151
152        loop {
153            target += dir;
154            if target < 0 || target >= 63 {
155                break;
156            }
157
158            let target_row = target / 8;
159            let target_col = target % 8;
160            let from_row = from as i8 / 8;
161            let from_col = from as i8 % 8;
162
163            if (target_row - from_row).abs() != (target_col - from_col).abs() {
164                break; // not diagonal anymore → wrapped
165            }
166
167            let spotlight = 1u64 << target;
168            // friendly piece? stop
169            if (position.bb_sides[friendly_index].0 & spotlight) != 0 {
170                break;
171            }
172            // enemy piece? push move, then stop
173            if (position.bb_sides[enemy_index].0 & spotlight) != 0 {
174                moves.push(Move {
175                    from,
176                    to: target as u8,
177                    piece,
178                    promoted_from_pawn: false,
179                });
180                break;
181            }
182            moves.push(Move {
183                from,
184                to: target as u8,
185                piece,
186                promoted_from_pawn: false,
187            });
188        }
189    }
190    moves
191}

198    let mut moves = Vec::new();
199

201
202    let directions: [i8; 4] = [1, 8, -1, -8];
203    for &dir in &directions {
204        let mut target = from as i8;
205
206        loop {
207            target += dir;
208            if target < 0 || target > 63 {
209                break;
210            }
211
212            let target_row = target / 8;
213            let target_col = target % 8;
214            let from_row = from as i8 / 8;
215            let from_col = from as i8 % 8;
216
217            if target_row != from_row && target_col != from_col {
218                break;
219            }

221            // friendly piece: stop

223                break;
224            }
225            // enemy piece: push move, then stop
226            if (position.bb_sides[enemy_index].0 & spotlight) != 0 {
227                moves.push(Move {
228                    from,
229                    to: target as u8,
230                    piece,
231                    promoted_from_pawn: false,
232                });
233                break;
234            }
235            moves.push(Move {

237                to: target as u8,
238                piece,
239                promoted_from_pawn: false,
240            });
241        }
242    }

246/// Generates pseudo-legal queen moves from a square.
247///
248/// Queens combine rook and bishop moves.
249pub fn valid_queen_moves(from: u8, piece: Piece, position: &Position) -> Vec<Move> {
250    let mut moves = valid_rook_moves(from, piece, position);
251    moves.extend(valid_bishop_moves(from, piece, position));
252    moves

256/// Generates pseudo-legal pawn moves from a square.
257///
258/// Handles:
259/// - Single and double forward pushes
260/// - Diagonal captures
261/// - En passant captures
262/// - Promotions at the last rank
263pub fn valid_pawn_moves(from: u8, piece: Piece, position: &Position) -> Vec<Move> {
264    let mut moves = Vec::new();
265
266    let (dir, start_row) = match piece {
267        Piece::Pawn(Color::White) => (8, 1),  // white moves up
268        Piece::Pawn(Color::Black) => (-8, 6), // black moves down
269        _ => return moves,
270    };
271    let (_friendly_index, enemy_index) = piece_indexes(piece);
272
273    // reducing this repetition?
274    let from_row = (from / 8) as i8;
275    let from_col = (from % 8) as i8;
276
277    let forward1 = from as i8 + dir;
278    let forward1_mask = 1u64 << forward1;
279    if (position.bb_sides[0].0 & forward1_mask == 0)
280        && (position.bb_sides[1].0 & forward1_mask == 0)
281    {
282        if is_pawn_promotion(forward1 as u8, piece) {
283            for promoted_piece in valid_pawn_promotions(piece) {
284                moves.push(Move {
285                    from,
286                    to: forward1 as u8,
287                    piece: promoted_piece,
288                    promoted_from_pawn: true,
289                });
290            }
291        } else {
292            moves.push(Move {
293                from,
294                to: forward1 as u8,
295                piece,
296                promoted_from_pawn: false,
297            });
298        }
299
300        // 2-squares forward from starting row
301        if from_row == start_row {
302            let forward2 = forward1 + dir;
303            let forward2_mask = 1u64 << forward2;
304            if (position.bb_sides[0].0 & forward2_mask == 0)
305                && (position.bb_sides[1].0 & forward2_mask == 0)
306            {
307                // **also check intermediate square is empty**
308                if (position.bb_sides[0].0 & forward1_mask == 0)
309                    && (position.bb_sides[1].0 & forward1_mask == 0)
310                {
311                    moves.push(Move {
312                        from,
313                        to: forward2 as u8,
314                        piece,
315                        promoted_from_pawn: false,
316                    });
317                }
318            }
319        }
320    }
321    // diagonal moves
322    for &diag in &[dir - 1, dir + 1] {
323        let target = from as i8 + diag;
324        if target < 0 || target > 63 {
325            continue;
326        }
327        let target_col = target % 8;
328        if (target_col - from_col).abs() != 1 {
329            continue;
330        }
331
332        let target_mask = 1u64 << target;
333        if (position.bb_sides[enemy_index].0 & target_mask != 0)
334            || Some(target as u8) == position.en_passant
335        {
336            if is_pawn_promotion(target as u8, piece) {
337                for promoted_piece in valid_pawn_promotions(piece) {
338                    moves.push(Move {
339                        from,
340                        to: target as u8,
341                        piece: promoted_piece,
342                        promoted_from_pawn: true,
343                    });
344                }
345            } else {
346                moves.push(Move {
347                    from,
348                    to: target as u8,
349                    piece,
350                    promoted_from_pawn: false,
351                });
352            }
353        }
354        // en passant square check
355        if let Some(ep_square) = position.en_passant {
356            if ep_square as i8 == target {
357                moves.push(Move {
358                    from,
359                    to: target as u8,
360                    piece,
361                    promoted_from_pawn: false,
362                });
363            }
364        }
365        // indicating in move that move is a capture?
366    }
367    moves
368}

370/// Generates pseudo-legal king moves from a square.
371///
372/// Handles:
373/// - Single-square king moves in all directions
374/// - Excludes moves landing on friendly pieces
375/// - Adds possible castling moves via [`castling_moves`]
376///
377/// Does not check for moving into check.
378pub fn valid_king_moves(from: u8, piece: Piece, position: &Position) -> Vec<Move> {
379    let mut moves = Vec::new();
380
381    let (own_index, _enemy_index) = piece_indexes(piece);
382    let directions: [i8; 8] = [1, -1, 8, -8, 7, -7, 9, -9];
383
384    let from_row = (from / 8) as i8;
385    let from_col = (from % 8) as i8;
386
387    for &dir in &directions {
388        let target = from as i8 + dir;
389
390        if target < 0 || target > 63 {
391            continue;
392        }
393
394        let target_row = target / 8;
395        let target_col = target % 8;
396
397        if dir == 1 || dir == -1 {
398            // must stay in same row
399            if target_row != from_row {
400                continue;
401            }
402        } else if dir == 7 || dir == -7 || dir == 9 || dir == -9 {
403            // must stay diagonal
404            if (target_row - from_row).abs() != (target_col - from_col).abs() {
405                continue;
406            }
407        }
408
409        let spotlight = 1u64 << target;
410
411        // cannot land on own piece
412        if (position.bb_sides[own_index].0 & spotlight) != 0 {
413            continue;
414        }
415
416        moves.push(Move {
417            from,
418            to: target as u8,
419            piece,
420            promoted_from_pawn: false,
421        });
422    }
423    moves.extend(castling_moves(from, piece, position));
424    moves
